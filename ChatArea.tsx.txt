
import React, { useState, useRef, useEffect } from 'react';
import { Message, Mode } from '../types';

interface ChatAreaProps {
  messages: Message[];
  isTyping: boolean;
  onSendMessage: (content: string) => void;
  mode: Mode;
}

const ChatArea: React.FC<ChatAreaProps> = ({ messages, isTyping, onSendMessage, mode }) => {
  const [input, setInput] = useState('');
  const scrollRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages, isTyping]);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (!input.trim() || isTyping) return;
    onSendMessage(input);
    setInput('');
  };

  return (
    <div className="flex-1 flex flex-col h-full relative overflow-hidden">
      <div ref={scrollRef} className="flex-1 overflow-y-auto px-6 md:px-12 py-10 space-y-8 max-w-4xl mx-auto w-full custom-scroll scroll-smooth">
        {messages.length === 0 ? (
          <div className="h-full flex flex-col items-center justify-center text-center space-y-10 animate-fadeIn">
            <div className={`w-24 h-24 rounded-[2rem] flex items-center justify-center shadow-2xl animate-float ${mode === 'owner' ? 'bg-black border-2 border-gold text-gold shadow-gold/10' : 'bg-indigo-600 text-white'}`}>
              <svg className="w-12 h-12" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={1.5} d="M13 10V3L4 14h7v7l9-11h-7z" /></svg>
            </div>
            <div>
              <h2 className="text-5xl md:text-7xl font-black tracking-tighter mb-4">
                {mode === 'owner' ? <span className="text-gold">BOSS MODE</span> : 'Ramzz'}<span className="opacity-10">.ai</span>
              </h2>
              <p className="text-slate-500 text-lg font-medium italic">Ready to cook something genius, bestie?</p>
            </div>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4 w-full">
              {['Gimana cara dapet passive income di 2025?', 'Coding landing page luxury pake React', 'Bikin rencana diet tapi tetep makan enak', 'Jelasin Web3 ke gw yang masih awam'].map(q => (
                <button key={q} onClick={() => onSendMessage(q)} className="p-6 glass-panel rounded-3xl text-sm font-bold text-left hover:border-indigo-500/50 transition-all hover:scale-105 active:scale-95">"{q}"</button>
              ))}
            </div>
          </div>
        ) : (
          messages.map((m, idx) => (
            <div key={idx} className={`flex w-full ${m.role === 'user' ? 'justify-end' : 'justify-start'} animate-slideUp`}>
              <div className={`max-w-[85%] flex gap-4 ${m.role === 'user' ? 'flex-row-reverse' : 'flex-row'}`}>
                <div className={`w-10 h-10 rounded-2xl flex-shrink-0 flex items-center justify-center text-[10px] font-black border ${m.role === 'user' ? 'bg-indigo-600 border-white/20' : (mode === 'owner' ? 'bg-black border-gold text-gold' : 'bg-slate-800 border-white/10')}`}>
                  {m.role === 'user' ? 'ME' : 'RZ'}
                </div>
                <div className={`p-5 rounded-3xl shadow-xl text-[16px] leading-relaxed ${m.role === 'user' ? 'bg-indigo-600 text-white rounded-tr-none' : `glass-panel text-slate-200 rounded-tl-none border-l-4 ${mode === 'owner' ? 'border-l-gold' : 'border-l-indigo-500'}`}`}>
                  {m.content}
                </div>
              </div>
            </div>
          ))
        )}
        {isTyping && (
          <div className="flex justify-start gap-4">
            <div className="w-10 h-10 rounded-2xl bg-slate-800 flex items-center justify-center"><div className="w-2 h-2 bg-indigo-500 rounded-full animate-ping"></div></div>
            <div className="glass-panel px-6 py-4 rounded-3xl flex gap-1.5 items-center">
              <div className="w-1.5 h-1.5 bg-indigo-500 rounded-full animate-bounce"></div>
              <div className="w-1.5 h-1.5 bg-indigo-500 rounded-full animate-bounce [animation-delay:0.2s]"></div>
              <div className="w-1.5 h-1.5 bg-indigo-500 rounded-full animate-bounce [animation-delay:0.4s]"></div>
            </div>
          </div>
        )}
      </div>

      <div className="p-8 md:p-12 relative z-20">
        <div className="max-w-4xl mx-auto">
          <form onSubmit={handleSubmit} className={`glass-panel p-2 rounded-[2.5rem] border-2 transition-all duration-500 flex items-center ${mode === 'owner' ? 'border-gold/30 shadow-gold/5' : 'border-white/5 focus-within:border-indigo-500/50'}`}>
            <input type="text" value={input} onChange={(e) => setInput(e.target.value)} placeholder={mode === 'owner' ? "Perintah Anda, Boss?" : "Tanya Ramzz apapun..."} className="flex-1 bg-transparent px-6 py-4 outline-none text-white font-medium text-lg placeholder-slate-700" />
            <button type="submit" disabled={!input.trim() || isTyping} className={`p-5 rounded-3xl transition-all ${input.trim() && !isTyping ? (mode === 'owner' ? 'bg-gold text-black' : 'bg-indigo-600 text-white') : 'bg-white/5 text-slate-700'}`}>
              <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M14 5l7 7-7 7M5 12h14" /></svg>
            </button>
          </form>
          <p className="text-center text-[10px] text-slate-700 mt-6 font-bold uppercase tracking-[0.3em] opacity-40">Ramzz Engine v6 - Ultra Low Latency Activated</p>
        </div>
      </div>
    </div>
  );
};

export default ChatArea;
